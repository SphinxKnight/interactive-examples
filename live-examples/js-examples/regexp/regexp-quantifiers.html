var ghostSpeak = "booh boooooooh";
var regexpSpooky = /bo{3,}h/;
console.log(ghostSpeak.match(regexpSpooky));
// expected output: boooooooh

var htmlSnippet = "<span foo='bar' baz>Hello<span/>"
var regexpSpan = /<span.*?>|<span\/>/g;
console.log(htmlSnippet.replace(regexpSpan, ""));
// expected output: Hello
