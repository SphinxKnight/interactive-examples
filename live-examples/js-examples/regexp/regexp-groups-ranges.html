var aliceExcerpt = "The Caterpillar and Alice looked at each other";
var regexpWithoutE = /\b[a-df-z]+\b/ig;
console.log(aliceExcerpt.match(regexpWithoutE));
// expected output [ 'and', 'at']

var webHistory = "The Web is born on the 03/12/1989.";
var regexpDate = /([0-9]{2})\/([0-9]{2})\/([0-9]{4})/;
var match = webHistory.match(regexpDate);
console.log("Year:", match[3], ", month:", match[1], ", day:", match[2]);
// expected output "Year: 1989 , month: 03 , day: 12"
